[ IF ]
DECLARE
	a, b, c(1:4), d
BEGIN
	a ASSIGN 8;
	b ASSIGN 9;
	d ASSIGN 1;
	
	IF a LEQ b THEN
		WRITE 10;

		IF a LE b THEN
			WRITE 20;

			IF b GE a THEN
				WRITE 30;
			ENDIF

			IF b GEQ a THEN
				WRITE 40;
			ENDIF
		ELSE
			WRITE -20;

			IF b GE a THEN
				WRITE 8;
			ENDIF

		ENDIF

	ELSE
		WRITE -10;
	ENDIF

	WRITE 0;
END
